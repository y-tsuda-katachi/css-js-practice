<details>
<summary>解答例</summary>

```JS
const loadingLabel = document.querySelector("#loading-label");
const images = document.querySelectorAll(".rand-img");

setTimeout(() => {
    loadingLabel.style.display = "none";
    for (let i = 0; i < images.length; i++) {
        const keyframes = {
            opacity: [0, 1],
        };
        const options = {
            duration: 600,
            delay: i * 300,
            fill: "forwards",
        };
        images[i].animate(keyframes, options);
    }
}, 3000); // 画像が読み込まれるまでの待機時間
```
