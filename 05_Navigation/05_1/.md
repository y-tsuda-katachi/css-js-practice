<details>
<summary>解答例</summary>

```JS
const menuItems = document.querySelectorAll(".menu li:has(> .sub-menu)");

menuItems.forEach((item) => {
    item.addEventListener("click", (e) => {
        e.preventDefault();

        const subMenu = item.querySelector(".sub-menu");
        if (subMenu) {
            subMenu.classList.toggle("active");
        }
    })
});
```
